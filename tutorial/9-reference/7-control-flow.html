Control Flow

If / Else

if x > 10 {
    print "big"
} else {
    print "small"
}

While Loop

x = 0
while x < 5 {
    print x
    x = x + 1
}

For Loop

fruits = ["apple", "banana", "cherry"]
for fruit in fruits {
    print fruit
}

Break & Continue

for i in [1,2,3,4,5] {
    if i == 3 {
        continue
    }
    if i == 5 {
        break
    }
    print i
}
